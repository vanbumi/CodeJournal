# Tips N Tricks

### Add item into list by **$scope. .push**

	$scope.addPost = function(){
		if(!$scope.title || $scope.title === '') {return;}
		$scope.posts.push({
	    title: $scope.title,
	    link: $scope.link,
	    upvotes: 0
	  });
		$scope.title = '';
	  $scope.link = '';
	};
